//extend方法


function extend(o,p){

for(prop in p){

o[prop]=p[prop];

}

return o;


}

//union方法


function union(o,p){

return extend(extend({},o),p);


}

//任意数量数组合并

someArray.reduce(union)

//查找一个数组中含有该元素的所有索引


function findAll(a,x){

var len = a.length,

pos = 0,

res = [];

while(pos<len){

pos = a.indexOf(x,pos);

if(pos==-1) break;

res.push(pos);

pos+=1;

}

return res;

}

findAll([1,2,3,55,2,1,2,3,4,1],1)

//[0, 5, 9]
